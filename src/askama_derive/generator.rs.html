25pub(crate) fn template_to_string(
26    buf: &mut Buffer,
27    input: &TemplateInput<'_>,
28    contexts: &HashMap<&Arc<Path>, Context<'_>, FxBuildHasher>,
29    heritage: Option<&Heritage<'_, '_>>,
30    tmpl_kind: TmplKind<'_>,
31) -> Result<usize, CompileError> {
32    let generator = Generator::new(
33        input,
34        contexts,
35        heritage,
36        MapChain::default(),
37        input.block.is_some(),
38        0,
39    );
40    let size_hint = match generator.impl_template(buf, tmpl_kind) {
41        Err(mut err) if err.span.is_none() => {
42            err.span = input.source_span;
43            Err(err)
44        }
45        result => result,
46    }?;
47
48    if tmpl_kind == TmplKind::Struct {
49        impl_everything(input.ast, buf);
50    }
51    Ok(size_hint)
52}

55pub(crate) enum TmplKind<'a> {
56    /// [`askama::Template`]
57    Struct,
58    /// [`askama::helpers::EnumVariantTemplate`]
59    Variant,
60    /// Used in `blocks` implementation
61    #[allow(unused)]

63}

65struct Generator<'a, 'h> {
66    /// The template input state: original struct AST and attributes
67    input: &'a TemplateInput<'a>,
68    /// All contexts, keyed by the package-relative template path
69    contexts: &'a HashMap<&'a Arc<Path>, Context<'a>, FxBuildHasher>,
70    /// The heritage contains references to blocks and their ancestry
71    heritage: Option<&'h Heritage<'a, 'h>>,
72    /// Variables accessible directly from the current scope (not redirected to context)
73    locals: MapChain<'a>,
74    /// Suffix whitespace from the previous literal. Will be flushed to the
75    /// output buffer unless suppressed by whitespace suppression on the next
76    /// non-literal.
77    next_ws: Option<&'a str>,
78    /// Whitespace suppression from the previous non-literal. Will be used to
79    /// determine whether to flush prefix whitespace from the next literal.
80    skip_ws: Whitespace,
81    /// If currently in a block, this will contain the name of a potential parent block
82    super_block: Option<(&'a str, usize)>,
83    /// Buffer for writable
84    buf_writable: WritableBuffer<'a>,
85    /// Used in blocks to check if we are inside a filter block.
86    is_in_filter_block: usize,
87    /// Set of called macros we are currently in. Used to prevent (indirect) recursions.
88    seen_macros: Vec<(&'a Macro<'a>, Option<FileInfo<'a>>)>,
89}

117    // Implement `Template` for the given context struct.
118    fn impl_template(
119        mut self,
120        buf: &mut Buffer,
121        tmpl_kind: TmplKind<'a>,
122    ) -> Result<usize, CompileError> {
123        let ctx = &self.contexts[&self.input.path];
124
125        let target = match tmpl_kind {
126            TmplKind::Struct => "askama::Template",
127            TmplKind::Variant => "askama::helpers::EnumVariantTemplate",
128            TmplKind::Block(trait_name) => trait_name,
129        };
130        write_header(self.input.ast, buf, target);
131        buf.write(
132            "fn render_into_with_values<AskamaW>(\
133                &self,\
134                __askama_writer: &mut AskamaW,\
135                __askama_values: &dyn askama::Values\
136            ) -> askama::Result<()>\
137            where \
138                AskamaW: askama::helpers::core::fmt::Write + ?askama::helpers::core::marker::Sized\
139            {\
140                #[allow(unused_imports)]\
141                use askama::{\
142                    filters::{AutoEscape as _, WriteWritable as _},\
143                    helpers::{ResultConverter as _, core::fmt::Write as _},\
144                };",
145        );
146
147        if let Some(full_config_path) = &self.input.config.full_config_path {
148            buf.write(format_args!(
149                "const _: &[askama::helpers::core::primitive::u8] =\
150                askama::helpers::core::include_bytes!({:?});",
151                full_config_path.display()
152            ));
153        }
154
155        // Make sure the compiler understands that the generated code depends on the template files.
156        let mut paths = self
157            .contexts
158            .keys()
159            .map(|path| -> &Path { path })
160            .collect::<Vec<_>>();
161        paths.sort();
162        for path in paths {
163            // Skip the fake path of templates defined in rust source.
164            let path_is_valid = match self.input.source {
165                Source::Path(_) => true,
166                Source::Source(_) => path != &*self.input.path,
167            };
168            if path_is_valid {
169                buf.write(format_args!(
170                    "const _: &[askama::helpers::core::primitive::u8] =\
171                        askama::helpers::core::include_bytes!({:#?});",
172                    path.canonicalize().as_deref().unwrap_or(path),
173                ));
174            }
175        }
176
177        let size_hint = self.impl_template_inner(ctx, buf)?;
178
179        buf.write("askama::Result::Ok(()) }");
180        if tmpl_kind == TmplKind::Struct {
181            buf.write(format_args!(
182                "const SIZE_HINT: askama::helpers::core::primitive::usize = {size_hint}usize;",
183            ));
184        }
185
186        buf.write('}');
187
188        #[cfg(feature = "blocks")]
189        for block in self.input.blocks {
190            self.impl_block(buf, block)?;
191        }
192
193        Ok(size_hint)
194    }

196    #[cfg(feature = "blocks")]
197    fn impl_block(
198        &self,
199        buf: &mut Buffer,
200        block: &crate::input::Block,
201    ) -> Result<(), CompileError> {
202        // RATIONALE: `*self` must be the input type, implementation details should not leak:
203        // - impl Self { fn as_block(self) } ->
204        // - struct __Askama__Self__as__block__Wrapper { this: self } ->
205        // - impl Template for __Askama__Self__as__block__Wrapper { fn render_into_with_values() } ->
206        // - impl __Askama__Self__as__block for Self { render_into_with_values() }
207
208        use quote::quote_spanned;
209        use syn::{GenericParam, Ident, Lifetime, LifetimeParam, Token};
210
211        let span = block.span;
212        buf.write(
213            "\
214            #[allow(missing_docs, non_camel_case_types, non_snake_case, unreachable_pub)]\
215            const _: () = {",
216        );
217
218        let ident = &self.input.ast.ident;
219
220        let doc = format!(
221            "A sub-template that renders only the block `{}` of [`{ident}`].",
222            block.name
223        );
224        let method_name = format!("as_{}", block.name);
225        let trait_name = format!("__Askama__{ident}__as__{}", block.name);
226        let wrapper_name = format!("__Askama__{ident}__as__{}__Wrapper", block.name);
227        let self_lt_name = format!("'__Askama__{ident}__as__{}__self", block.name);
228
229        let method_id = Ident::new(&method_name, span);
230        let trait_id = Ident::new(&trait_name, span);
231        let wrapper_id = Ident::new(&wrapper_name, span);
232        let self_lt = Lifetime::new(&self_lt_name, span);
233
234        // generics of the input with an additional lifetime to capture `self`
235        let mut wrapper_generics = self.input.ast.generics.clone();
236        if wrapper_generics.lt_token.is_none() {
237            wrapper_generics.lt_token = Some(Token![<](span));
238            wrapper_generics.gt_token = Some(Token![>](span));
239        }
240        wrapper_generics.params.insert(
241            0,
242            GenericParam::Lifetime(LifetimeParam::new(self_lt.clone())),
243        );
244
245        let (impl_generics, ty_generics, where_clause) = self.input.ast.generics.split_for_impl();
246        let (wrapper_impl_generics, wrapper_ty_generics, wrapper_where_clause) =
247            wrapper_generics.split_for_impl();
248
249        let input = TemplateInput {
250            block: Some((&block.name, span)),
251            blocks: &[],
252            ..self.input.clone()
253        };
254        let size_hint = template_to_string(
255            buf,
256            &input,
257            self.contexts,
258            self.heritage,
259            TmplKind::Block(&trait_name),
260        )?;
261
262        buf.write(quote_spanned! {
263            span =>
264            pub trait #trait_id {
265                fn render_into_with_values<AskamaW>(
266                    &self,
267                    writer: &mut AskamaW,
268                    values: &dyn askama::Values,
269                ) -> askama::Result<()>
270                where
271                    AskamaW:
272                        askama::helpers::core::fmt::Write + ?askama::helpers::core::marker::Sized;
273            }
274
275            impl #impl_generics #ident #ty_generics #where_clause {
276                #[inline]
277                #[doc = #doc]
278                pub fn #method_id(&self) -> impl askama::Template + '_ {
279                    #wrapper_id {
280                        this: self,
281                    }
282                }
283            }
284
285            #[askama::helpers::core::prelude::rust_2021::derive(
286                askama::helpers::core::prelude::rust_2021::Clone,
287                askama::helpers::core::prelude::rust_2021::Copy
288            )]
289            pub struct #wrapper_id #wrapper_generics #wrapper_where_clause {
290                this: &#self_lt #ident #ty_generics,
291            }
292
293            impl #wrapper_impl_generics askama::Template
294            for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
295                #[inline]
296                fn render_into_with_values<AskamaW>(
297                    &self,
298                    writer: &mut AskamaW,
299                    values: &dyn askama::Values
300                ) -> askama::Result<()>
301                where
302                    AskamaW: askama::helpers::core::fmt::Write + ?askama::helpers::core::marker::Sized
303                {
304                    <_ as #trait_id>::render_into_with_values(self.this, writer, values)
305                }
306
307                const SIZE_HINT: askama::helpers::core::primitive::usize = #size_hint;
308            }
309
310            // cannot use `crate::integrations::impl_fast_writable()` w/o cloning the struct
311            impl #wrapper_impl_generics askama::FastWritable
312            for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
313                #[inline]
314                fn write_into<AskamaW>(
315                    &self,
316                    dest: &mut AskamaW,
317                    values: &dyn askama::Values,
318                ) -> askama::Result<()>
319                where
320                    AskamaW: askama::helpers::core::fmt::Write + ?askama::helpers::core::marker::Sized
321                {
322                    <_ as askama::Template>::render_into_with_values(self, dest, values)
323                }
324            }
325
326            // cannot use `crate::integrations::impl_display()` w/o cloning the struct
327            impl #wrapper_impl_generics askama::helpers::core::fmt::Display
328            for #wrapper_id #wrapper_ty_generics #wrapper_where_clause {
329                #[inline]
330                fn fmt(
331                    &self,
332                    f: &mut askama::helpers::core::fmt::Formatter<'_>
333                ) -> askama::helpers::core::fmt::Result {
334                    <_ as askama::Template>::render_into(self, f)
335                        .map_err(|_| askama::helpers::core::fmt::Error)
336                }
337            }
338        });
339
340        buf.write("};");
341        Ok(())
342    }

372/// In here, we inspect in the expression if it is a literal, and if it is, whether it
373/// can be escaped at compile time.
374fn compile_time_escape<'a>(expr: &Expr<'a>, escaper: &str) -> Option<Writable<'a>> {
375    // we only optimize for known escapers
376    enum OutputKind {
377        Html,
378        Text,
379    }
380
381    // we only optimize for known escapers
382    let output = match escaper.strip_prefix("askama::filters::")? {
383        "Html" => OutputKind::Html,
384        "Text" => OutputKind::Text,
385        _ => return None,
386    };
387
388    // for now, we only escape strings, chars, numbers, and bools at compile time
389    let value = match *expr {
390        Expr::StrLit(StrLit {
391            prefix: None,
392            content,
393        }) => {
394            if content.find('\\').is_none() {
395                // if the literal does not contain any backslashes, then it does not need unescaping
396                Cow::Borrowed(content)
397            } else {
398                // the input could be string escaped if it contains any backslashes
399                let input = format!(r#""{content}""#);
400                let input = input.parse().ok()?;
401                let input = syn::parse2::<syn::LitStr>(input).ok()?;
402                Cow::Owned(input.value())
403            }
404        }
405        Expr::CharLit(CharLit {
406            prefix: None,
407            content,
408        }) => {
409            if content.find('\\').is_none() {
410                // if the literal does not contain any backslashes, then it does not need unescaping
411                Cow::Borrowed(content)
412            } else {
413                // the input could be string escaped if it contains any backslashes
414                let input = format!(r#"'{content}'"#);
415                let input = input.parse().ok()?;
416                let input = syn::parse2::<syn::LitChar>(input).ok()?;
417                Cow::Owned(input.value().to_string())
418            }
419        }
420        Expr::NumLit(_, value) => {
421            enum NumKind {
422                Int(Option<IntKind>),
423                Float(Option<FloatKind>),
424            }
425
426            let (orig_value, kind) = match value {
427                Num::Int(value, kind) => (value, NumKind::Int(kind)),
428                Num::Float(value, kind) => (value, NumKind::Float(kind)),
429            };
430            let value = match orig_value.chars().any(|c| c == '_') {
431                true => Cow::Owned(orig_value.chars().filter(|&c| c != '_').collect()),
432                false => Cow::Borrowed(orig_value),
433            };
434
435            fn int<T: ToString, E>(
436                from_str_radix: impl Fn(&str, u32) -> Result<T, E>,
437                value: &str,
438            ) -> Option<String> {
439                Some(from_str_radix(value, 10).ok()?.to_string())
440            }
441
442            let value = match kind {
443                NumKind::Int(Some(IntKind::I8)) => int(i8::from_str_radix, &value)?,
444                NumKind::Int(Some(IntKind::I16)) => int(i16::from_str_radix, &value)?,
445                NumKind::Int(Some(IntKind::I32)) => int(i32::from_str_radix, &value)?,
446                NumKind::Int(Some(IntKind::I64)) => int(i64::from_str_radix, &value)?,
447                NumKind::Int(Some(IntKind::I128)) => int(i128::from_str_radix, &value)?,
448                NumKind::Int(Some(IntKind::Isize)) => int(TargetIsize::from_str_radix, &value)?,
449                NumKind::Int(Some(IntKind::U8)) => int(u8::from_str_radix, &value)?,
450                NumKind::Int(Some(IntKind::U16)) => int(u16::from_str_radix, &value)?,
451                NumKind::Int(Some(IntKind::U32)) => int(u32::from_str_radix, &value)?,
452                NumKind::Int(Some(IntKind::U64)) => int(u64::from_str_radix, &value)?,
453                NumKind::Int(Some(IntKind::U128)) => int(u128::from_str_radix, &value)?,
454                NumKind::Int(Some(IntKind::Usize)) => int(TargetUsize::from_str_radix, &value)?,
455                NumKind::Int(None) => match value.starts_with('-') {
456                    true => int(i128::from_str_radix, &value)?,
457                    false => int(u128::from_str_radix, &value)?,
458                },
459                NumKind::Float(Some(FloatKind::F32)) => value.parse::<f32>().ok()?.to_string(),
460                NumKind::Float(Some(FloatKind::F64) | None) => {
461                    value.parse::<f64>().ok()?.to_string()
462                }
463                // FIXME: implement once `f16` and `f128` are available
464                NumKind::Float(Some(FloatKind::F16 | FloatKind::F128)) => return None,
465            };
466            match value == orig_value {
467                true => Cow::Borrowed(orig_value),
468                false => Cow::Owned(value),
469            }
470        }
471        Expr::BoolLit(true) => Cow::Borrowed("true"),
472        Expr::BoolLit(false) => Cow::Borrowed("false"),
473        _ => return None,
474    };
475
476    // escape the un-string-escaped input using the selected escaper
477    Some(Writable::Lit(match output {
478        OutputKind::Text => value,
479        OutputKind::Html => {
480            let mut escaped = String::with_capacity(value.len() + 20);
481            write_escaped_str(&mut escaped, &value).ok()?;
482            match escaped == value {
483                true => value,
484                false => Cow::Owned(escaped),
485            }
486        }
487    }))
488}

512struct MapChain<'a> {
513    scopes: Vec<HashMap<Cow<'a, str>, LocalMeta, FxBuildHasher>>,
514}
515
516impl<'a> MapChain<'a> {
517    fn new_empty() -> Self {
518        Self { scopes: vec![] }
519    }
520
521    /// Iterates the scopes in reverse and returns `Some(LocalMeta)`
522    /// from the first scope where `key` exists.
523    fn get<'b>(&'b self, key: &str) -> Option<&'b LocalMeta> {
524        self.scopes.iter().rev().find_map(|set| set.get(key))
525    }
526
527    fn is_current_empty(&self) -> bool {
528        self.scopes.last().unwrap().is_empty()
529    }
530
531    fn insert(&mut self, key: Cow<'a, str>, val: LocalMeta) {
532        self.scopes.last_mut().unwrap().insert(key, val);
533
534        // Note that if `insert` returns `Some` then it implies
535        // an identifier is reused. For e.g. `{% macro f(a, a) %}`
536        // and `{% let (a, a) = ... %}` then this results in a
537        // generated template, which when compiled fails with the
538        // compile error "identifier `a` used more than once".
539    }
540
541    fn insert_with_default(&mut self, key: Cow<'a, str>) {
542        self.insert(key, LocalMeta::default());
543    }
544
545    fn resolve(&self, name: &str) -> Option<String> {
546        let name = normalize_identifier(name);
547        self.get(&Cow::Borrowed(name)).map(|meta| match &meta.refs {
548            Some(expr) => expr.clone(),
549            None => name.to_string(),
550        })
551    }
552
553    fn resolve_or_self(&self, name: &str) -> String {
554        let name = normalize_identifier(name);
555        self.resolve(name).unwrap_or_else(|| format!("self.{name}"))
556    }
557}

559impl Default for MapChain<'_> {
560    fn default() -> Self {
561        Self {
562            scopes: vec![HashMap::default()],
563        }
564    }

567/// Returns `true` if enough assumptions can be made,
568/// to determine that `self` is copyable.
569fn is_copyable(expr: &Expr<'_>) -> bool {
570    is_copyable_within_op(expr, false)
571}
572
573fn is_copyable_within_op(expr: &Expr<'_>, within_op: bool) -> bool {
574    match expr {
575        Expr::BoolLit(_)
576        | Expr::NumLit(_, _)
577        | Expr::StrLit(_)
578        | Expr::CharLit(_)
579        | Expr::BinOp(_, _, _) => true,
580        Expr::Unary(.., expr) => is_copyable_within_op(expr, true),
581        Expr::Range(..) => true,
582        // The result of a call likely doesn't need to be borrowed,
583        // as in that case the call is more likely to return a
584        // reference in the first place then.
585        Expr::Call { .. } | Expr::Path(..) | Expr::Filter(..) | Expr::RustMacro(..) => true,
586        // If the `expr` is within a `Unary` or `BinOp` then
587        // an assumption can be made that the operand is copy.
588        // If not, then the value is moved and adding `.clone()`
589        // will solve that issue. However, if the operand is
590        // implicitly borrowed, then it's likely not even possible
591        // to get the template to compile.
592        _ => within_op && is_attr_self(expr),
593    }
594}

596/// Returns `true` if this is an `Attr` where the `obj` is `"self"`.
597fn is_attr_self(mut expr: &Expr<'_>) -> bool {
598    loop {
599        match expr {
600            Expr::Attr(obj, _) if matches!(***obj, Expr::Var("self")) => return true,
601            Expr::Attr(obj, _) if matches!(***obj, Expr::Attr(..)) => expr = obj,
602            _ => return false,
603        }
604    }
605}

621impl<'a> WritableBuffer<'a> {
622    fn push(&mut self, writable: Writable<'a>) {
623        if !self.discard {
624            self.buf.push(writable);
625        }
626    }
627}

643/// Identifiers to be replaced with raw identifiers, so as to avoid
644/// collisions between template syntax and Rust's syntax. In particular
645/// [Rust keywords](https://doc.rust-lang.org/reference/keywords.html)
646/// should be replaced, since they're not reserved words in Askama
647/// syntax but have a high probability of causing problems in the
648/// generated code.
649///
650/// This list excludes the Rust keywords *self*, *Self*, and *super*
651/// because they are not allowed to be raw identifiers, and *loop*
652/// because it's used something like a keyword in the template
653/// language.
654fn normalize_identifier(ident: &str) -> &str {
655    // This table works for as long as the replacement string is the original string
656    // prepended with "r#". The strings get right-padded to the same length with b'_'.
657    // While the code does not need it, please keep the list sorted when adding new
658    // keywords.
659
660    if ident.len() > MAX_RUST_KEYWORD_LEN {
661        return ident;
662    }
663    let kws = RUST_KEYWORDS[ident.len()];
664
665    let mut padded_ident = [0; MAX_RUST_KEYWORD_LEN];
666    padded_ident[..ident.len()].copy_from_slice(ident.as_bytes());
667
668    // Since the individual buckets are quite short, a linear search is faster than a binary search.
669    for probe in kws {
670        if padded_ident == *AsciiChar::slice_as_bytes(probe[2..].try_into().unwrap()) {
671            return AsciiStr::from_slice(&probe[..ident.len() + 2]);
672        }
673    }
674    ident
675}
