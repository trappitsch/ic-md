85impl TryFrom<dsl_hir::GlobalConfigList> for mir::GlobalConfig {
86    type Error = syn::Error;
87
88    fn try_from(value: dsl_hir::GlobalConfigList) -> Result<Self, Self::Error> {
89        let mut global_config = mir::GlobalConfig::default();
90
91        for config in value.configs.iter() {
92            let same_config_count = value
93                .configs
94                .iter()
95                .filter(|check_config| {
96                    std::mem::discriminant(*check_config) == std::mem::discriminant(config)
97                })
98                .count();
99
100            if same_config_count > 1 {
101                return Err(syn::Error::new(
102                    Span::call_site(),
103                    format!("Duplicate global config found: `{config:?}`"),
104                ));
105            }
106
107            match config.clone() {
108                dsl_hir::GlobalConfig::DefaultRegisterAccess(value) => {
109                    global_config.default_register_access = value.into()
110                }
111                dsl_hir::GlobalConfig::DefaultFieldAccess(value) => {
112                    global_config.default_field_access = value.into()
113                }
114                dsl_hir::GlobalConfig::DefaultBufferAccess(value) => {
115                    global_config.default_buffer_access = value.into()
116                }
117                dsl_hir::GlobalConfig::DefaultByteOrder(value) => {
118                    global_config.default_byte_order = Some(value.into())
119                }
120                dsl_hir::GlobalConfig::DefaultBitOrder(value) => {
121                    global_config.default_bit_order = value.into()
122                }
123                dsl_hir::GlobalConfig::RegisterAddressType(value) => {
124                    global_config.register_address_type = Some(value.try_into()?)
125                }
126                dsl_hir::GlobalConfig::CommandAddressType(value) => {
127                    global_config.command_address_type = Some(value.try_into()?)
128                }
129                dsl_hir::GlobalConfig::BufferAddressType(value) => {
130                    global_config.buffer_address_type = Some(value.try_into()?)
131                }
132                dsl_hir::GlobalConfig::NameWordBoundaries(value) => {
133                    global_config.name_word_boundaries = value
134                }
135                dsl_hir::GlobalConfig::DefmtFeature(lit_str) => {
136                    global_config.defmt_feature = Some(lit_str.value())
137                }
138            }
139        }
140
141        Ok(global_config)
142    }
143}

145fn get_description(attrs: &dsl_hir::AttributeList) -> Option<String> {
146    let str = attrs
147        .attributes
148        .iter()
149        .filter_map(|attr| match attr {
150            dsl_hir::Attribute::Doc(val) => Some(val.as_str()),
151            dsl_hir::Attribute::Cfg(_, _) => None,
152        })
153        .collect::<Vec<_>>()
154        .join("\n");
155
156    if str.is_empty() { None } else { Some(str) }

159fn get_cfg_attr(attrs: &dsl_hir::AttributeList) -> Result<mir::Cfg, syn::Error> {
160    let mut cfg_attrs = attrs
161        .attributes
162        .iter()
163        .filter_map(|attr| match attr {
164            dsl_hir::Attribute::Cfg(val, span) => Some((val, span)),
165            dsl_hir::Attribute::Doc(_) => None,
166        })
167        .collect::<Vec<_>>();
168
169    match cfg_attrs.len() {
170        0 => Ok(mir::Cfg::new(None)),
171        1 => Ok(mir::Cfg::new(Some(&cfg_attrs.remove(0).0.clone()))),
172        n => Err(syn::Error::new(
173            *cfg_attrs.remove(1).1,
174            format!("Only one cfg attribute is allowed, but {n} are found"),
175        )),
176    }
177}

179fn transform_object_list(
180    list: dsl_hir::ObjectList,
181    global_config: &mir::GlobalConfig,
182) -> Result<Vec<mir::Object>, syn::Error> {
183    let mut objects = Vec::new();
184
185    for object in list.objects.into_iter() {
186        let object = match object {
187            dsl_hir::Object::Block(block) => {
188                mir::Object::Block(transform_block(block, global_config)?)
189            }
190            dsl_hir::Object::Register(register) => {
191                mir::Object::Register(transform_register(register, global_config)?)
192            }
193            dsl_hir::Object::Command(command) => {
194                mir::Object::Command(transform_command(command, global_config)?)
195            }
196            dsl_hir::Object::Buffer(buffer) => {
197                mir::Object::Buffer(transform_buffer(buffer, global_config)?)
198            }
199            dsl_hir::Object::Ref(ref_object) => mir::Object::Ref(transform_ref(ref_object)?),
200        };
201
202        objects.push(object);
203    }
204
205    Ok(objects)
206}

367fn transform_command(
368    command: dsl_hir::Command,
369    global_config: &mir::GlobalConfig,
370) -> Result<mir::Command, syn::Error> {
371    let command_value = command.value.ok_or_else(|| {
372        syn::Error::new(
373            command.identifier.span(),
374            format!("Command `{}` must have a value", command.identifier),
375        )
376    })?;
377    Ok(mir::Command {
378        cfg_attr: get_cfg_attr(&command.attribute_list)?,
379        description: get_description(&command.attribute_list).unwrap_or_default(),
380        name: command.identifier.to_string(),
381        address: match &command_value {
382            dsl_hir::CommandValue::Basic(lit) => lit,
383            dsl_hir::CommandValue::Extended {
384                command_item_list, ..
385            } => command_item_list
386                .items
387                .iter()
388                .find_map(|item| match item {
389                    dsl_hir::CommandItem::Address(lit) => Some(lit),
390                    _ => None,
391                })
392                .ok_or_else(|| {
393                    syn::Error::new(
394                        command.identifier.span(),
395                        format!("Command `{}` must have an address", command.identifier),
396                    )
397                })?,
398        }
399        .base10_parse()?,
400        byte_order: match &command_value {
401            dsl_hir::CommandValue::Basic(_) => None,
402            dsl_hir::CommandValue::Extended {
403                command_item_list, ..
404            } => command_item_list.items.iter().find_map(|item| match item {
405                dsl_hir::CommandItem::ByteOrder(order) => Some((*order).into()),
406                _ => None,
407            }),
408        },
409        bit_order: match &command_value {
410            dsl_hir::CommandValue::Basic(_) => None,
411            dsl_hir::CommandValue::Extended {
412                command_item_list, ..
413            } => command_item_list.items.iter().find_map(|item| match item {
414                dsl_hir::CommandItem::BitOrder(order) => Some((*order).into()),
415                _ => None,
416            }),
417        }
418        .unwrap_or(global_config.default_bit_order),
419        allow_bit_overlap: match &command_value {
420            dsl_hir::CommandValue::Basic(_) => None,
421            dsl_hir::CommandValue::Extended {
422                command_item_list, ..
423            } => command_item_list.items.iter().find_map(|item| match item {
424                dsl_hir::CommandItem::AllowBitOverlap(b) => Some(b.value),
425                _ => None,
426            }),
427        }
428        .unwrap_or_default(),
429        allow_address_overlap: match &command_value {
430            dsl_hir::CommandValue::Basic(_) => None,
431            dsl_hir::CommandValue::Extended {
432                command_item_list, ..
433            } => command_item_list.items.iter().find_map(|item| match item {
434                dsl_hir::CommandItem::AllowAddressOverlap(b) => Some(b.value),
435                _ => None,
436            }),
437        }
438        .unwrap_or_default(),
439        size_bits_in: match &command_value {
440            dsl_hir::CommandValue::Basic(_) => None,
441            dsl_hir::CommandValue::Extended {
442                command_item_list, ..
443            } => command_item_list.items.iter().find_map(|item| match item {
444                dsl_hir::CommandItem::SizeBitsIn(size) => Some(size.base10_parse()),
445                _ => None,
446            }),
447        }
448        .unwrap_or(Ok(0))?,
449        size_bits_out: match &command_value {
450            dsl_hir::CommandValue::Basic(_) => None,
451            dsl_hir::CommandValue::Extended {
452                command_item_list, ..
453            } => command_item_list.items.iter().find_map(|item| match item {
454                dsl_hir::CommandItem::SizeBitsOut(size) => Some(size.base10_parse()),
455                _ => None,
456            }),
457        }
458        .unwrap_or(Ok(0))?,
459        repeat: match &command_value {
460            dsl_hir::CommandValue::Basic(_) => None,
461            dsl_hir::CommandValue::Extended {
462                command_item_list, ..
463            } => command_item_list.items.iter().find_map(|item| match item {
464                dsl_hir::CommandItem::Repeat(repeat) => Some(repeat.clone().try_into()),
465                _ => None,
466            }),
467        }
468        .transpose()?,
469        in_fields: match &command_value {
470            dsl_hir::CommandValue::Basic(_)
471            | dsl_hir::CommandValue::Extended {
472                in_field_list: None,
473                ..
474            } => Vec::new(),
475            dsl_hir::CommandValue::Extended {
476                in_field_list: Some(in_field_list),
477                ..
478            } => in_field_list
479                .fields
480                .iter()
481                .map(|field| transform_field(field, global_config))
482                .collect::<Result<_, _>>()?,
483        },
484        out_fields: match &command_value {
485            dsl_hir::CommandValue::Basic(_)
486            | dsl_hir::CommandValue::Extended {
487                out_field_list: None,
488                ..
489            } => Vec::new(),
490            dsl_hir::CommandValue::Extended {
491                out_field_list: Some(out_field_list),
492                ..
493            } => out_field_list
494                .fields
495                .iter()
496                .map(|field| transform_field(field, global_config))
497                .collect::<Result<_, _>>()?,
498        },
499    })
500}

502fn transform_field(
503    field: &dsl_hir::Field,
504    global_config: &mir::GlobalConfig,
505) -> Result<mir::Field, syn::Error> {
506    let field_cfg_attr = get_cfg_attr(&field.attribute_list)?;
507    let field_description = get_description(&field.attribute_list).unwrap_or_default();
508
509    Ok(mir::Field {
510        cfg_attr: field_cfg_attr.clone(),
511        description: field_description.clone(),
512        name: field.identifier.to_string(),
513        access: field
514            .access
515            .map(Into::into)
516            .unwrap_or(global_config.default_field_access),
517        base_type: field.base_type.into(),
518        field_conversion: field
519            .field_conversion
520            .as_ref()
521            .map(|fc| transform_field_conversion(field_description, fc))
522            .transpose()?,
523        field_address: match &field.field_address {
524            dsl_hir::FieldAddress::Integer(start) if field.base_type.is_bool() => {
525                start.base10_parse()?..start.base10_parse()?
526            }
527            dsl_hir::FieldAddress::Integer(_) => {
528                return Err(syn::Error::new(
529                    field.identifier.span(),
530                    format!(
531                        "Field `{}` has a non-bool base type and must specify the start and the end address",
532                        field.identifier
533                    ),
534                ));
535            }
536            dsl_hir::FieldAddress::Range { start, end } => {
537                start.base10_parse()?..end.base10_parse()?
538            }
539            dsl_hir::FieldAddress::RangeInclusive { start, end } => {
540                start.base10_parse()?..(end.base10_parse::<u32>()? + 1)
541            }
542        },
543    })
544}

546fn transform_field_conversion(
547    field_description: String,
548    field_conversion: &dsl_hir::FieldConversion,
549) -> Result<mir::FieldConversion, syn::Error> {
550    match field_conversion {
551        dsl_hir::FieldConversion::Direct { path, use_try } => Ok(mir::FieldConversion::Direct {
552            type_name: path
553                .to_token_stream()
554                .to_string()
555                .replace(char::is_whitespace, ""),
556            use_try: *use_try,
557        }),
558        dsl_hir::FieldConversion::Enum {
559            identifier,
560            enum_variant_list,
561            use_try,
562        } => Ok(mir::FieldConversion::Enum {
563            enum_value: mir::Enum::new(
564                field_description,
565                identifier.to_string(),
566                enum_variant_list
567                    .variants
568                    .iter()
569                    .map(|v| {
570                        Ok(mir::EnumVariant {
571                            cfg_attr: get_cfg_attr(&v.attribute_list)?,
572                            description: get_description(&v.attribute_list).unwrap_or_default(),
573                            name: v.identifier.to_string(),
574                            value: match &v.enum_value {
575                                None => mir::EnumValue::Unspecified,
576                                Some(dsl_hir::EnumValue::Specified(val)) => {
577                                    mir::EnumValue::Specified(val.base10_parse()?)
578                                }
579                                Some(dsl_hir::EnumValue::Default) => mir::EnumValue::Default,
580                                Some(dsl_hir::EnumValue::CatchAll) => mir::EnumValue::CatchAll,
581                            },
582                        })
583                    })
584                    .collect::<Result<_, syn::Error>>()?,
585            ),
586            use_try: *use_try,
587        }),
588    }
589}

649fn transform_block_override(
650    block_override: dsl_hir::Block,
651) -> Result<mir::BlockOverride, syn::Error> {
652    if !block_override.attribute_list.attributes.is_empty() {
653        return Err(syn::Error::new(
654            block_override.identifier.span(),
655            "No attributes (cfg or doc) are allowed on block overrides",
656        ));
657    }
658
659    if !block_override.object_list.objects.is_empty() {
660        return Err(syn::Error::new(
661            block_override.identifier.span(),
662            "No objects may be defined on block overrides",
663        ));
664    }
665
666    Ok(mir::BlockOverride {
667        name: block_override.identifier.to_string(),
668        address_offset: block_override
669            .block_item_list
670            .block_items
671            .iter()
672            .find_map(|item| match item {
673                dsl_hir::BlockItem::AddressOffset(address_offset) => {
674                    Some(address_offset.base10_parse())
675                }
676                _ => None,
677            })
678            .transpose()?,
679        repeat: block_override
680            .block_item_list
681            .block_items
682            .iter()
683            .find_map(|item| match item {
684                dsl_hir::BlockItem::Repeat(repeat) => Some(repeat.clone().try_into()),
685                _ => None,
686            })
687            .transpose()?,
688    })

691fn transform_register_override(
692    register_override: dsl_hir::Register,
693) -> Result<mir::RegisterOverride, syn::Error> {
694    if !register_override.attribute_list.attributes.is_empty() {
695        return Err(syn::Error::new(
696            register_override.identifier.span(),
697            "No attributes (cfg or doc) are allowed on register overrides",
698        ));
699    }
700
701    if !register_override.field_list.fields.is_empty() {
702        return Err(syn::Error::new(
703            register_override.identifier.span(),
704            "No fields are allowed on register overrides",
705        ));
706    }
707
708    for item in register_override.register_item_list.register_items.iter() {
709        match item {
710            dsl_hir::RegisterItem::ByteOrder(_) => {
711                return Err(syn::Error::new(
712                    register_override.identifier.span(),
713                    "No `ByteOrder` is allowed on register overrides",
714                ));
715            }
716            dsl_hir::RegisterItem::BitOrder(_) => {
717                return Err(syn::Error::new(
718                    register_override.identifier.span(),
719                    "No `BitOrder` is allowed on register overrides",
720                ));
721            }
722            dsl_hir::RegisterItem::SizeBits(_) => {
723                return Err(syn::Error::new(
724                    register_override.identifier.span(),
725                    "No `SizeBits` is allowed on register overrides",
726                ));
727            }
728            dsl_hir::RegisterItem::AllowBitOverlap(_) => {
729                return Err(syn::Error::new(
730                    register_override.identifier.span(),
731                    "No `AllowBitOverlap` is allowed on register overrides",
732                ));
733            }
734            dsl_hir::RegisterItem::Access(_) => {}
735            dsl_hir::RegisterItem::Address(_) => {}
736            dsl_hir::RegisterItem::ResetValueInt(_) => {}
737            dsl_hir::RegisterItem::ResetValueArray(_) => {}
738            dsl_hir::RegisterItem::Repeat(_) => {}
739            dsl_hir::RegisterItem::AllowAddressOverlap(_) => {}
740        }
741    }
742
743    Ok(mir::RegisterOverride {
744        name: register_override.identifier.to_string(),
745        access: register_override
746            .register_item_list
747            .register_items
748            .iter()
749            .find_map(|i| match i {
750                dsl_hir::RegisterItem::Access(access) => Some((*access).into()),
751                _ => None,
752            }),
753        address: register_override
754            .register_item_list
755            .register_items
756            .iter()
757            .find_map(|i| match i {
758                dsl_hir::RegisterItem::Address(addr) => Some(addr.base10_parse()),
759                _ => None,
760            })
761            .transpose()?,
762        allow_address_overlap: register_override
763            .register_item_list
764            .register_items
765            .iter()
766            .find_map(|i| match i {
767                dsl_hir::RegisterItem::AllowAddressOverlap(b) => Some(b.value),
768                _ => None,
769            })
770            .unwrap_or_default(),
771        reset_value: register_override
772            .register_item_list
773            .register_items
774            .iter()
775            .find_map(|i| match i {
776                dsl_hir::RegisterItem::ResetValueArray(arr) => {
777                    Some(Ok(mir::ResetValue::Array(arr.clone())))
778                }
779                dsl_hir::RegisterItem::ResetValueInt(int) => Some(
780                    int.base10_parse::<i128>()
781                        .map(|v| v as u128)
782                        .or_else(|_| int.base10_parse::<u128>())
783                        .map_err(|e| {
784                            syn::Error::new(
785                                int.span(),
786                                format!("{e}: number is parsed as an i128 or u128"),
787                            )
788                        })
789                        .map(mir::ResetValue::Integer),
790                ),
791                _ => None,
792            })
793            .transpose()?,
794        repeat: register_override
795            .register_item_list
796            .register_items
797            .iter()
798            .find_map(|i| match i {
799                dsl_hir::RegisterItem::Repeat(repeat) => Some(repeat.clone().try_into()),
800                _ => None,
801            })
802            .transpose()?,
803    })
804}

806fn transform_command_override(
807    command_override: dsl_hir::Command,
808) -> Result<mir::CommandOverride, syn::Error> {
809    if !command_override.attribute_list.attributes.is_empty() {
810        return Err(syn::Error::new(
811            command_override.identifier.span(),
812            "No attributes (cfg or doc) are allowed on command overrides",
813        ));
814    }
815
816    let command_item_list = match &command_override.value {
817        Some(dsl_hir::CommandValue::Extended {
818            in_field_list: Some(_),
819            ..
820        }) => {
821            return Err(syn::Error::new(
822                command_override.identifier.span(),
823                "No `in` field list is allowed on command overrides",
824            ));
825        }
826        Some(dsl_hir::CommandValue::Extended {
827            out_field_list: Some(_),
828            ..
829        }) => {
830            return Err(syn::Error::new(
831                command_override.identifier.span(),
832                "No `out` field list is allowed on command overrides",
833            ));
834        }
835        Some(dsl_hir::CommandValue::Extended {
836            command_item_list,
837            in_field_list: None,
838            out_field_list: None,
839        }) => {
840            for ci in command_item_list.items.iter() {
841                match ci {
842                    dsl_hir::CommandItem::ByteOrder(_) => {
843                        return Err(syn::Error::new(
844                            command_override.identifier.span(),
845                            "No `ByteOrder` is allowed on command overrides",
846                        ));
847                    }
848                    dsl_hir::CommandItem::BitOrder(_) => {
849                        return Err(syn::Error::new(
850                            command_override.identifier.span(),
851                            "No `BitOrder` is allowed on command overrides",
852                        ));
853                    }
854                    dsl_hir::CommandItem::SizeBitsIn(_) => {
855                        return Err(syn::Error::new(
856                            command_override.identifier.span(),
857                            "No `SizeBitsIn` is allowed on command overrides",
858                        ));
859                    }
860                    dsl_hir::CommandItem::SizeBitsOut(_) => {
861                        return Err(syn::Error::new(
862                            command_override.identifier.span(),
863                            "No `SizeBitsOut` is allowed on command overrides",
864                        ));
865                    }
866                    dsl_hir::CommandItem::AllowBitOverlap(_) => {
867                        return Err(syn::Error::new(
868                            command_override.identifier.span(),
869                            "No `AllowBitOverlap` is allowed on command overrides",
870                        ));
871                    }
872                    dsl_hir::CommandItem::AllowAddressOverlap(_) => {}
873                    dsl_hir::CommandItem::Repeat(_) => {}
874                    dsl_hir::CommandItem::Address(_) => {}
875                }
876            }
877
878            command_item_list
879        }
880        Some(dsl_hir::CommandValue::Basic(_)) => {
881            return Err(syn::Error::new(
882                command_override.identifier.span(),
883                "No basic address specifier is allowed on command overrides. Use the extended syntax with `{ }` instead",
884            ));
885        }
886        None => {
887            return Err(syn::Error::new(
888                command_override.identifier.span(),
889                "A value is required on command overrides",
890            ));
891        }
892    };
893
894    Ok(mir::CommandOverride {
895        name: command_override.identifier.to_string(),
896        address: command_item_list
897            .items
898            .iter()
899            .find_map(|item| match item {
900                dsl_hir::CommandItem::Address(lit) => Some(lit),
901                _ => None,
902            })
903            .map(|lit| lit.base10_parse())
904            .transpose()?,
905        allow_address_overlap: command_item_list
906            .items
907            .iter()
908            .find_map(|item| match item {
909                dsl_hir::CommandItem::AllowAddressOverlap(lit) => Some(lit.value),
910                _ => None,
911            })
912            .unwrap_or_default(),
913        repeat: command_item_list
914            .items
915            .iter()
916            .find_map(|item| match item {
917                dsl_hir::CommandItem::Repeat(repeat) => Some(mir::Repeat::try_from(repeat.clone())),
918                _ => None,
919            })
920            .transpose()?,
921    })
922}
