92pub enum FieldConversionMethod {
93    None,
94    Into(String),
95    UnsafeInto(String),
96    TryInto(String),
97    Bool,
98}
99
100impl FieldConversionMethod {
101    pub fn conversion_type(&self) -> Option<&String> {
102        match self {
103            FieldConversionMethod::None => None,
104            FieldConversionMethod::Into(type_path) => Some(type_path),
105            FieldConversionMethod::UnsafeInto(type_path) => Some(type_path),
106            FieldConversionMethod::TryInto(type_path) => Some(type_path),
107            FieldConversionMethod::Bool => None,
108        }
109    }
110}
