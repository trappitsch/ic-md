6pub fn run_pass(device: &mut Device) -> anyhow::Result<()> {
7    let root_block = device
8        .blocks
9        .iter()
10        .find(|b| b.root)
11        .expect("There's always a root block");
12
13    let claimed_addresses = get_block_claimed_addresses(device, root_block, 0, &[])?;
14
15    for (i, claimed_address) in claimed_addresses.iter().enumerate() {
16        for other_claimed_address in claimed_addresses.get(i + 1..).unwrap_or_default() {
17            let types_same = claimed_address.address_type == other_claimed_address.address_type;
18            let both_allow_overlap =
19                claimed_address.allow_overlap && other_claimed_address.allow_overlap;
20            let address_same = claimed_address.address == other_claimed_address.address;
21
22            if address_same && types_same && !both_allow_overlap {
23                let mut name0 = claimed_address.name.clone();
24                if let Some(repeat_index) = claimed_address.repeat_index {
25                    name0 += &format!(" (index: {repeat_index})");
26                }
27                let mut name1 = other_claimed_address.name.clone();
28                if let Some(repeat_index) = other_claimed_address.repeat_index {
29                    name1 += &format!(" (index: {repeat_index})");
30                }
31                bail!(
32                    "Objects \"{name0}\" and \"{name1}\" use the same address ({}). If this is intended, then allow address overlap on both objects.",
33                    claimed_address.address
34                );
35            }
36        }
37    }
38
39    Ok(())
40}

42fn get_block_claimed_addresses(
43    device: &Device,
44    block: &Block,
45    current_address_offset: i64,
46    name_stack: &[String],
47) -> anyhow::Result<Vec<ClaimedAddress>> {
48    let mut claimed_adresses = Vec::new();
49
50    for method in &block.methods {
51        let current_address_offset =
52            current_address_offset + method.address.to_string().parse::<i64>().unwrap();
53
54        let (repeat_count, repeat_stride, repeat): (i64, i64, bool) = match &method.kind {
55            BlockMethodKind::Normal => (1, 0, false),
56            BlockMethodKind::Repeated { count, stride } => (
57                count.to_string().parse().unwrap(),
58                stride.to_string().parse().unwrap(),
59                true,
60            ),
61        };
62
63        let claimed_address_type = match &method.method_type {
64            BlockMethodType::Block { name } => {
65                let sub_block = device
66                    .blocks
67                    .iter()
68                    .find(|b| b.name == *name)
69                    .expect("There's always a root block");
70
71                for i in 0..repeat_count {
72                    let next_name_stack = name_stack
73                        .iter()
74                        .cloned()
75                        .chain([format!(
76                            "{} (index: {i})",
77                            name.to_string().to_case(Case::Pascal)
78                        )])
79                        .collect::<Vec<_>>();
80
81                    claimed_adresses.extend(get_block_claimed_addresses(
82                        device,
83                        sub_block,
84                        current_address_offset + i * repeat_stride,
85                        &next_name_stack,
86                    )?);
87                }
88                continue;
89            }
90            BlockMethodType::Register { .. } => ClaimedAddressType::Register,
91            BlockMethodType::Command { .. } => ClaimedAddressType::Command,
92            BlockMethodType::Buffer { .. } => ClaimedAddressType::Buffer,
93        };
94
95        for i in 0..repeat_count {
96            use itertools::Itertools;
97
98            claimed_adresses.push(ClaimedAddress {
99                name: name_stack
100                    .iter()
101                    .cloned()
102                    .chain([method.name.to_string().to_case(Case::Pascal)])
103                    .join("::"),
104                repeat_index: repeat.then_some(i),
105                address: current_address_offset + i * repeat_stride,
106                allow_overlap: method.allow_address_overlap,
107                address_type: claimed_address_type,
108            });
109        }
110    }
111
112    Ok(claimed_adresses)
113}

116enum ClaimedAddressType {

118    Command,
