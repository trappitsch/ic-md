9pub fn generate_block(
10    value: &Block,
11    internal_address_type: &Ident,
12    register_address_type: &Ident,
13) -> TokenStream {
14    let Block {
15        cfg_attr,
16        doc_attr,
17        root,
18        name,
19        methods,
20    } = value;
21
22    let (generics, interface_declaration, address_param, address_specifier, interface_borrow) =
23        if *root {
24            (
25                quote! { I },
26                quote! { I },
27                None,
28                quote! { 0 },
29                quote! { &mut self.interface },
30            )
31        } else {
32            (
33                quote! { 'i, I },
34                quote! { &'i mut I },
35                Some(quote! { base_address: #internal_address_type }),
36                quote! { base_address },
37                quote! { self.interface },
38            )
39        };
40
41    let method_impls = methods
42        .iter()
43        .map(|m| generate_method(m, internal_address_type));
44
45    let (new_hidden_if_not_root, new_access, new_const) = if *root {
46        (quote! {}, quote! { pub }, quote! { const })
47    } else {
48        (quote! { #[doc(hidden)] }, quote! {}, quote! {})
49    };
50
51    let generate_read_all_registers_items = |use_async: bool| {
52        methods
53            .iter()
54            .filter_map(move |m| match &m.method_type {
55                BlockMethodType::Register {
56                    access: mir::Access::RO | mir::Access::RW,
57                    ..
58                } => {
59                    let register_name = &m.name;
60                    let address = &m.address;
61                    let cfg_attr = &m.cfg_attr;
62
63                    let read_function = match use_async {
64                        true => quote! { read_async().await },
65                        false => quote! { read() },
66                    };
67
68                    let (count, stride, index_required) = match &m.kind {
69                        BlockMethodKind::Normal => (1i64, Literal::i64_unsuffixed(0), false),
70                        BlockMethodKind::Repeated { count, stride } => {
71                            (count.to_string().parse().unwrap(), stride.clone(), true)
72                        }
73                    };
74
75                    Some((0..count).map(move |index| {
76                    let (index_param, register_display_name) = match index_required {
77                        true => {
78                            let index_param = Literal::i64_unsuffixed(index);
79                            (quote! { #index_param }, format!("{}[{index}]", m.name))
80                        }
81                        false => (quote! {}, m.name.to_string()),
82                    };
83                    let index = Literal::i64_unsuffixed(index);
84                    quote! {
85                        #cfg_attr
86                        let reg = self.#register_name(#index_param).#read_function?;
87                        #cfg_attr
88                        callback(#address + #index * #stride, #register_display_name, reg.into());
89                    }
90                }))
91                }
92                _ => None,
93            })
94            .flatten()
95    };
96
97    let read_all_registers_items = generate_read_all_registers_items(false);
98    let read_async_all_registers_items = generate_read_all_registers_items(true);
99
100    let read_all_docs = quote! {
101        /// Read all readable register values in this block from the device.
102        /// The callback is called for each of them.
103        /// Any registers in child blocks are not included.
104        ///
105        /// The callback has three arguments:
106        ///
107        /// - The address of the register
108        /// - The name of the register (with index for repeated registers)
109        /// - The read value from the register
110        ///
111        /// This is useful for e.g. debug printing all values.
112        /// The given [field_sets::FieldSetValue] has a Debug and Format implementation that forwards to the concrete type
113        /// the lies within so it can be printed without matching on it.
114    };
115
116    quote! {
117        #doc_attr
118        #cfg_attr
119        #[derive(Debug)]
120        pub struct #name<#generics> {
121            pub(crate) interface: #interface_declaration,
122            #[doc(hidden)]
123            base_address: #internal_address_type,
124        }
125
126        #cfg_attr
127        impl<#generics> #name<#generics> {
128            /// Create a new instance of the block based on device interface
129            #new_hidden_if_not_root
130            #new_access #new_const fn new(interface: #interface_declaration, #address_param) -> Self {
131                Self {
132                    interface,
133                    base_address: #address_specifier,
134                }
135            }
136
137            /// A reference to the interface used to communicate with the device
138            pub(crate) fn interface(&mut self) -> &mut I {
139                #interface_borrow
140            }
141
142            #read_all_docs
143            pub fn read_all_registers(
144                &mut self,
145                mut callback: impl FnMut(#register_address_type, &'static str, field_sets::FieldSetValue)
146            ) -> Result<(), I::Error>
147                where I: ::device_driver::RegisterInterface<AddressType = #register_address_type>
148            {
149                #(#read_all_registers_items)*
150                Ok(())
151            }
152
153            #read_all_docs
154            pub async fn read_all_registers_async(
155                &mut self,
156                mut callback: impl FnMut(#register_address_type, &'static str, field_sets::FieldSetValue)
157            ) -> Result<(), I::Error>
158                where I: ::device_driver::AsyncRegisterInterface<AddressType = #register_address_type>
159            {
160                #(#read_async_all_registers_items)*
161                Ok(())
162            }
163
164            #(#method_impls)*
165        }
166    }
167}

169fn generate_method(method: &BlockMethod, internal_address_type: &Ident) -> TokenStream {
170    let BlockMethod {
171        cfg_attr,
172        doc_attr,
173        name,
174        address,
175        allow_address_overlap: _,
176        kind,
177        method_type,
178    } = method;
179
180    let (return_type, address_conversion, default_arg) = match method_type {
181        BlockMethodType::Block { name } => (quote! { #name::<'_, I> }, quote! {}, quote! {}),
182        BlockMethodType::Register {
183            field_set_name,
184            access,
185            address_type,
186            reset_value_function: default_value_function_name,
187        } => (
188            quote! { ::device_driver::RegisterOperation::<'_, I, #address_type, field_sets::#field_set_name, ::device_driver::#access>  },
189            quote! { as #address_type },
190            quote! { , field_sets::#field_set_name::#default_value_function_name },
191        ),
192        BlockMethodType::Command {
193            field_set_name_in,
194            field_set_name_out,
195            address_type,
196        } => {
197            let field_set_name_in = match field_set_name_in {
198                Some(val) => quote! { field_sets::#val },
199                None => quote! { () },
200            };
201            let field_set_name_out = match field_set_name_out {
202                Some(val) => quote! { field_sets::#val },
203                None => quote! { () },
204            };
205            (
206                quote! { ::device_driver::CommandOperation::<'_, I, #address_type, #field_set_name_in, #field_set_name_out>  },
207                quote! { as #address_type },
208                quote! {},
209            )
210        }
211        BlockMethodType::Buffer {
212            access,
213            address_type,
214        } => (
215            quote! { ::device_driver::BufferOperation::<'_, I, #address_type, ::device_driver::#access>  },
216            quote! { as #address_type },
217            quote! {},
218        ),
219    };
220
221    let (index_param, address_calc, index_doc) = match kind {
222        BlockMethodKind::Normal => (None, quote! { self.base_address + #address }, None),
223        BlockMethodKind::Repeated { count, stride } => {
224            let doc = format!("Valid index range: 0..{count}");
225
226            let stride = stride.to_string().parse::<i64>().unwrap();
227
228            let operator = if stride.is_negative() {
229                quote! { - }
230            } else {
231                quote! { + }
232            };
233
234            let stride = Literal::u64_unsuffixed(stride.unsigned_abs());
235
236            (
237                Some(quote! { index: usize, }),
238                quote! { {
239                    assert!(index < #count);
240                    self.base_address + #address #operator index as #internal_address_type * #stride
241                } },
242                Some(quote! {
243                    #[doc = ""]
244                    #[doc = #doc]
245                }),
246            )
247        }
248    };
249
250    quote! {
251        #doc_attr
252        #index_doc
253        #cfg_attr
254        pub fn #name(&mut self, #index_param) -> #return_type {
255            let address = #address_calc;
256            #return_type::new(self.interface(), address #address_conversion #default_arg)
257        }
258    }
