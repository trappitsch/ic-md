5/// Propagate the cfg attributes. This makes sure that any child will be able to be active when the parent is not.
6///
7/// Currently this is done by `all(...)` combining the cfgs. In the future this should have a more advanced check.
8pub fn run_pass(device: &mut Device) -> anyhow::Result<()> {
9    let mut current_depth = 0;
10    let mut cfg_stack = vec![Cfg::new(None)];
11
12    recurse_objects_with_depth_mut(&mut device.objects, &mut |object, depth| {
13        if depth < current_depth {
14            cfg_stack.pop();
15            current_depth = depth;
16        }
17
18        let cfg_attr = object.cfg_attr_mut();
19        let new_cfg_attr = cfg_attr.combine(cfg_stack.last().unwrap());
20        *cfg_attr = new_cfg_attr.clone();
21
22        for field in object.field_sets_mut().flatten() {
23            // NOTE: We don't have to set the field cfg attr since it's part of the object
24            // We do need to update the enum though, since that's gonna be generated outside of the object
25            if let Some(FieldConversion::Enum { enum_value, .. }) = field.field_conversion.as_mut()
26            {
27                enum_value.cfg_attr = field.cfg_attr.combine(&new_cfg_attr);
28                // Just like we don't have to update the field cfg, we also don't have to update the enum variant cfgs
29            }
30        }
31
32        if object.as_block_mut().is_some() {
33            cfg_stack.push(new_cfg_attr);
34            current_depth += 1;
35        }
36
37        Ok(())
38    })
39}
