3//! Driver for the iC-MD quadrature counter.
4//! Built fully in Rust, uses [embedded_hal] and [device_driver].
5//!
6//! # Introduction
7//!
8//! The `IcMd` struct provides a high-level interface to interact with the iC-MD quadrature
9//! counter. However, you can also access the underlying device driver directly via the `device`
10//! field. Please read the device driver documentation for more information on what to expect
11//! when interfacing with the device driver directly.
12//! This low-level access is a temporary solution until the high-level interface is fully
13//! developped. When this well be the case is unclear. If you are interested in it, please let me
14//! know and I'm happy to prioritize the high-level features that are interesting to you.
15//!
16//! # Limitations
17//!
18//! The following features are currently only accessible via the low-level interface:
19//!
20//! - Reference register readout: It is unclear if this currently works, see code comment.
21//!
22//! The following features are currently not yet implemented:
23//!
24//! - Input configuration: You can currently not specify the input configuration.
25//!   - Differential or TTL inputs (Address 0x01, bit 7)
26//!   - Configuration to have Z signal clear counters 0 and/or 1 (Address 0x01, bits 5 and 6)
27//!   - Z signal configuration (Address 0x01, bits 3 and 4)
28//!   - Touch probe and AB registers (Address 0x01, bits 1 and 2)
29//!   - Differential input configuration selection (RS-422 (default) or LVDS) (Address 0x03, bit 7)
30//!

49/// The main driver struct of the crate representing the iC-MD quadrature counter.
50/// You can also access the underlying device driver directly via the `device` field.
51/// You are then yourself responsible for reading the correct counter configurations.
52#[derive(Debug)]
53pub struct IcMd<Spi> {
54    /// Provides acces to the underlying device driver.
55    pub device: Device<DeviceInterface<Spi>>,
56    /// Configuration of the counter, set only prior to calling `init()`.
57    counter_config: CntCfg,
58    /// Status of the device (error and warning flags). Read only, updated when reading the
59    /// counter.
60    device_status: DeviceStatus,
61    actuator_status: ActuatorStatus,
62}
63
64impl<Spi: SpiDevice> IcMd<Spi> {
65    /// Creates a new instance of the iC-MD driver.
66    /// By default, the counter is configured to 48-bit mode.
67    pub fn new(spi: Spi) -> Self {
68        Self {
69            device: Device::new(DeviceInterface::new(spi)),
70            counter_config: CntCfg::Cnt1Bit48(CntSetup::default()),
71            actuator_status: ActuatorStatus::default(),
72            device_status: DeviceStatus::default(),
73        }
74    }
75
76    /// Initialize the iC-MD device with the given configuration.
77    pub fn init(&mut self) -> Result<(), DeviceError<Spi::Error>> {
78        self.device
79            .counter_configuration()
80            .write(|reg| reg.set_value(self.counter_config.into()))?;
81
82        Ok(())
83    }
84
85    /// Set the actuator pins output to the given status.
86    /// Note that as far as the iC-MD is concerned, this status is "write only". Thus, there is no
87    /// function available to read the current status of the actuator pins. However, the stored
88    /// `actuator_status` variable will be updated according to what you set here.
89    ///
90    /// # Arguments
91    /// * `act0`: The status of actuator pin 0 (ACT0).
92    /// * `act1`: The status of actuator pin 1 (ACT1).
93    pub fn configure_actuator_pins(
94        &mut self,
95        act0: &PinStatus,
96        act1: &PinStatus,
97    ) -> Result<(), DeviceError<Spi::Error>> {
98        self.device.instruction_byte().write(|reg| {
99            reg.set_act_0(act0.into());
100            reg.set_act_1(act1.into());
101        })?;
102        self.actuator_status.act0 = *act0;
103        self.actuator_status.act1 = *act1;
104        Ok(())
105    }
106
107    /// Get current device status.
108    /// This is a cached value that is updated when reading the counter. It contains the error and
109    /// warning flags of the device. For a full device status, use `get_full_device_status()`.
110    pub fn get_device_status(&self) -> DeviceStatus {
111        self.device_status
112    }
113
114    /// Get the full device status by reading all the status registers.
115    /// This will reset many of the status bits to wait for the next event, problem, issue to
116    /// occur.
117    pub fn get_full_device_status(&mut self) -> Result<FullDeviceStatus, DeviceError<Spi::Error>> {
118        let status0 = self.device.status_0().read()?;
119        let status1 = self.device.status_1().read()?;
120        let status2 = self.device.status_2().read()?;
121
122        Ok(FullDeviceStatus {
123            cnt0_overflow: status0.ovf_0().into(),
124            cnt0_aberr: status0.ab_err_0().into(),
125            cnt0_zero: status0.zero_0().into(),
126            cnt1_overflow: status1.ovf_1().into(),
127            cnt1_aberr: status1.ab_err_1().into(),
128            cnt1_zero: status1.zero_1().into(),
129            cnt2_overflow: status2.ovf_2().into(),
130            cnt2_aberr: status2.ab_err_2().into(),
131            cnt2_zero: status2.zero_2().into(),
132            power_status: status0.p_dwn().into(),
133            ref_reg_status: status0.r_val().into(),
134            upd_reg_status: status0.upd_val().into(),
135            ref_cnt_status: status0.ovf_ref().into(),
136            ext_err_status: status1.ext_err().into(),
137            ext_warn_status: status1.ext_warn().into(),
138            comm_status: status1.com_col().into(),
139            tp_status: status0.tp_val().into(),
140            tpi_status: status1.tps().into(),
141            ssi_enabled: status2.en_ssi().into(),
142        })
143    }

145    /// Read the current counter value and return it.
146    pub fn read_counter(&mut self) -> Result<CntCount, DeviceError<Spi::Error>> {
147        match self.counter_config {
148            CntCfg::Cnt1Bit24(_) => {
149                let res = self.device.read_cnt_cfg_0().read()?;
150                self.set_device_status(res.nwarn(), res.nerr());
151                Ok(CntCount::Cnt1Bit24(res.cnt_0()))
152            }
153            CntCfg::Cnt2Bit24(_, _) => {
154                let res = self.device.read_cnt_cfg_1().read()?;
155                self.set_device_status(res.nwarn(), res.nerr());
156                Ok(CntCount::Cnt2Bit24(res.cnt_0(), res.cnt_1()))
157            }
158            CntCfg::Cnt1Bit48(_) => {
159                let res = self.device.read_cnt_cfg_2().read()?;
160                self.set_device_status(res.nwarn(), res.nerr());
161                Ok(CntCount::Cnt1Bit48(res.cnt_0()))
162            }
163            CntCfg::Cnt1Bit16(_) => {
164                let res = self.device.read_cnt_cfg_3().read()?;
165                self.set_device_status(res.nwarn(), res.nerr());
166                Ok(CntCount::Cnt1Bit16(res.cnt_0()))
167            }
168            CntCfg::Cnt1Bit32(_) => {
169                let res = self.device.read_cnt_cfg_4().read()?;
170                self.set_device_status(res.nwarn(), res.nerr());
171                Ok(CntCount::Cnt1Bit32(res.cnt_0()))
172            }
173            CntCfg::Cnt2Bit32Bit16(_, _) => {
174                let res = self.device.read_cnt_cfg_5().read()?;
175                self.set_device_status(res.nwarn(), res.nerr());
176                Ok(CntCount::Cnt2Bit32Bit16(res.cnt_0(), res.cnt_1()))
177            }
178            CntCfg::Cnt2Bit16(_, _) => {
179                let res = self.device.read_cnt_cfg_6().read()?;
180                self.set_device_status(res.nwarn(), res.nerr());
181                Ok(CntCount::Cnt2Bit16(res.cnt_0(), res.cnt_1()))
182            }
183            CntCfg::Cnt3Bit16(_, _, _) => {
184                let res = self.device.read_cnt_cfg_7().read()?;
185                self.set_device_status(res.nwarn(), res.nerr());
186                Ok(CntCount::Cnt3Bit16(res.cnt_0(), res.cnt_1(), res.cnt_2()))
187            }
188        }
189    }

191    /// Reset counters to zero.
192    /// You can select which counters should be set to zero using the specific arguments.
193    ///
194    /// # Arguments
195    /// * `cnt0`: If true, counter 0 is reset, else not.
196    /// * `cnt1`: If true, counter 1 is reset, else not.
197    /// * `cnt2`: If true, counter 2 is reset, else not.
198    pub fn reset_counters(
199        &mut self,
200        cnt0: bool,
201        cnt1: bool,
202        cnt2: bool,
203    ) -> Result<(), DeviceError<Spi::Error>> {
204        let act0 = &self.actuator_status.act0;
205        let act1 = &self.actuator_status.act1;
206        self.device.instruction_byte().write(|reg| {
207            reg.set_ab_res_0(cnt0);
208            reg.set_ab_res_1(cnt1);
209            reg.set_ab_res_2(cnt2);
210            reg.set_act_0(act0.into());
211            reg.set_act_1(act1.into());
212        })?;
213        Ok(())
214    }
215
216    /// Reset all counters.
217    /// Can be used to send reset commands to all counters.
218    pub fn reset_all_counters(&mut self) -> Result<(), DeviceError<Spi::Error>> {
219        self.reset_counters(true, true, true)?;
220        Ok(())
221    }
222
223    /// Touch probe instruction
224    /// Load touch probe 2 with touch probe 1 value and touch probe 1 wiht ABCNT value.
225    pub fn touch_probe_instruction(&mut self) -> Result<(), DeviceError<Spi::Error>> {
226        let act0 = &self.actuator_status.act0;
227        let act1 = &self.actuator_status.act1;
228        self.device.instruction_byte().write(|reg| {
229            reg.set_tp(true);
230            reg.set_act_0(act0.into());
231            reg.set_act_1(act1.into());
232        })?;
233        Ok(())
234    }
235
236    /// Set the counter configuration.
237    /// This should be done prior to calling `init()`.
238    pub fn set_counter_config(&mut self, config: CntCfg) {
239        self.counter_config = config;
240    }
241
242    /// Set device status from two bools that were read and passed on to here.
243    /// Note taat the inputs are from nerr and nwarn!
244    fn set_device_status(&mut self, nwarn: bool, nerr: bool) {
245        self.device_status.warning = match nwarn {
246            true => WarningStatus::Ok,
247            false => WarningStatus::Warning,
248        };
249        self.device_status.error = match nerr {
250            true => ErrorStatus::Ok,
251            false => ErrorStatus::Error,
252        };
253    }
